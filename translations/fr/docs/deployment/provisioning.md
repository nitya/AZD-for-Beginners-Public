  }
  properties: {
    reserved: true // Linux App Service Plan
  }
}

  name: '${applicationName}-web-${resourceToken}'
  location: location
  tags: tags
  properties: {
    serverFarmId: appServicePlan.id

      linuxFxVersion: 'NODE|18-lts'

          value: 'false'

          name: 'NODE_ENV'
          value: 'production'
